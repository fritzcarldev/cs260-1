Attribute Binding

Current State: {{state}} 

<div id="app">
  <p>
    Current State:
    <span v-bind:class="{hl: state}" v-bind:style="labelStyle">{{state}}</span>
  </p>
  <button v-bind:disabled="!state" v-on:click="toggle">false</button>
  <button :disabled="state" v-on:click="toggle">true</button>
</div>

<script>
  var app = new Vue({
    el: '#app',
    data: {
      state: true,
      labelStyle: {
        fontWeight: 'lighter',
      },
    },
    methods: {
      toggle: function () {
        this.state = !this.state;
        this.labelStyle.fontWeight = this.state ? 'bolder' : 'lighter';
      },
    },
  });
</script>
